cmd (terminal do Windows)
dir (trazer lista de diretórios)
cd / (base do diretório C)
cd Windows (foi para pastas Windows)
cd .. (voltar um ponto)
cls (limpa o terminal)
TAB (função de auto completar)
mkdir workspace (criar uma pasta "workspace")
echo hello (devolve o que digitou ex: hello)
echo hello > hello.txt (vê se o arquivo tem na pasta, se não tiver, ele mesmo cria)
del (deleta somente arquivos)
rmdir (deleta diretórios)
